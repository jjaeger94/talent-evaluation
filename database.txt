hier können die befehle für die Tabellen abgelegt werden

Tabelle jobs:

CREATE TABLE `commitIQ`.`jobs` (`ID` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL ,`location` VARCHAR(255) NOT NULL , `job_title` VARCHAR(255) NOT NULL , `added` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `edited` TIMESTAMP on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `criteria1` VARCHAR(255) NOT NULL , `criteria2` VARCHAR(255) NOT NULL , `criteria3` VARCHAR(255) NOT NULL , `completeness` TINYINT NOT NULL , `reference` TINYINT NOT NULL , `state` VARCHAR(255) NOT NULL DEFAULT 'active', PRIMARY KEY (`ID`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_bin;

Tabelle applications:

CREATE TABLE `commitIQ`.`applications` (`ID` INT NOT NULL AUTO_INCREMENT , `job_id` INT NOT NULL , `prename` VARCHAR(255) NOT NULL , `surname` VARCHAR(255) NOT NULL , `email` VARCHAR(255) NOT NULL , `salutation` INT NOT NULL , `classification` INT NOT NULL , `added` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `edited` TIMESTAMP on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `commitment_id` INT NOT NULL , `reference_id` INT NOT NULL , `filepath` VARCHAR(255) NOT NULL , PRIMARY KEY (`ID`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_bin;
