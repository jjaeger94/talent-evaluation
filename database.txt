Tabelle jobs:

CREATE TABLE `commitIQ`.`jobs` (`ID` INT NOT NULL AUTO_INCREMENT , `user_id` INT NOT NULL , `job_title` TEXT NOT NULL , `added` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `edited` TIMESTAMP on update CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP , `criteria1` TEXT NOT NULL , `criteria2` TEXT NOT NULL , `criteria3` TEXT NOT NULL , `completeness` TINYINT NOT NULL , `reference` TINYINT NOT NULL , PRIMARY KEY (`ID`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_bin;
